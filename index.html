// Configuración Básica const contractAddress = "0xc4158596B8F8E1cE337861dE0eCDc0c3045DbE88"; const usdtAddress = "0x55d398326f99059fF775485246999027B3197955"; // El ABI simplificado para el usuario (solo funciones de lectura y deposito) const abi = [ "function deposit(uint256 assets, address receiver) public returns (uint256)", "function getVaultDashboard(address _user) external view returns (uint256 capitalUsdt, uint256 residualLider, bool calificadoParaProfit, uint256 cronometroLock, uint256 sharePrice)", "function balanceOf(address account) public view returns (uint256)", "function approve(address spender, uint256 amount) public returns (bool)" ]; async function cargarDatosUsuario() { if (typeof window.ethereum !== 'undefined') { const provider = new ethers.BrowserProvider(window.ethereum); const signer = await provider.getSigner(); const address = await signer.getAddress(); const contract = new ethers.Contract(contractAddress, abi, signer); try { // Llamamos a la función mágica que creamos para el dashboard const data = await contract.getVaultDashboard(address); // Traduciendo los datos para mostrarlos en la web const balance = ethers.formatUnits(data.capitalUsdt, 18); const residual = ethers.formatUnits(data.residualLider, 18); const calificado = data.calificadoParaProfit ? "SÍ ✅" : "ESPERANDO (24h) ⏳"; const tiempoSegundos = Number(data.cronometroLock); const diasFaltantes = (tiempoSegundos / 86400).toFixed(1); // Inyectar estos valores en tu HTML document.getElementById('balanceText').innerText = `$${balance} USDT`; document.getElementById('residualText').innerText = `$${residual} USDT`; document.getElementById('statusText').innerText = calificado; document.getElementById('lockText').innerText = diasFaltantes + " días"; } catch (error) { console.error("Error al cargar datos:", error); } } }
